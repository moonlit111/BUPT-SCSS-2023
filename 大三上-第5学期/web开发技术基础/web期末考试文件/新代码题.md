### 第 1 题
```html
<body>
  <div id="d1">
    <p class="a">A1</p>
    <p class="a">A2</p>
    <p id="p3">A3</p>
    <p>A4</p>
    <p>A5</p>
  </div>
</body>
```
（1）第一步：设定 `#d1` 下所有 `p` 文本为红色（red）。
（2）第二步：设定 class 为 `a` 的段落为绿色（green）。
（3）第三步：用 CSS 组合选择器设定 `#p3` 之后所有 `p` 为橙色（orange）。
```js
var styleEl = document.createElement('style');
styleEl.type = 'text/css';

var cssText =
  '#d1 p { color: red; }\n' +
  '#d1 .a { color: green; }\n' +
  '#p3 ~ p { color: orange; }';

if (styleEl.styleSheet) {
  styleEl.styleSheet.cssText = cssText;
} else {
  styleEl.appendChild(document.createTextNode(cssText));
}

document.head.appendChild(styleEl);
```
### 第 2 题
```html
<body>
  <div id="d2">
    <span>one</span>
    <span class="k">two</span>
    <span class="k">three</span>
    <span id="s4">four</span>
    <span>five</span>
  </div>
</body>
```
（1）第一步：设定 `#d2` 下所有 `span` 文字为蓝色（blue）。
（2）第二步：设定 `.k` 为红色（red）。
（3）第三步：用相邻兄弟选择器，让紧跟在 `#s4` 后面的那个 `span` 为绿色（green）。
```js
var styleEl = document.createElement('style');
styleEl.type = 'text/css';

var cssText =
  '#d2 span { color: blue; }\n' +
  '#d2 .k { color: red; }\n' +
  '#s4 + span { color: green; }';

if (styleEl.styleSheet) {
  styleEl.styleSheet.cssText = cssText;
} else {
  styleEl.appendChild(document.createTextNode(cssText));
}

document.head.appendChild(styleEl);
```

```js
var cssText =

if (styleEl.styleSheet) {
  styleEl.styleSheet.cssText = cssText;
} else {
  styleEl.appendChild(document.createTextNode(cssText));

document.head.appendChild(styleEl);
```

```js
document.head.appendChild(styleEl);
```

### 第 6 题（伪元素 ::before）

（1）第一步：设定 `.item` 文字为灰色（gray）。
（2）第二步：设定 `#last` 为红色（red）。

```js
var styleEl = document.createElement('style');
styleEl.type = 'text/css';

var cssText =

  styleEl.styleSheet.cssText = cssText;

  styleEl.appendChild(document.createTextNode(cssText));

document.head.appendChild(styleEl);
```
